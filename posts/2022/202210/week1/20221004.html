                              class Solution {
public:
  ListNode *addTwoNumbers(ListNode *l1, ListNode *l2) {
    ListNode *dummy = new ListNode(-1);
    ListNode *curr = dummy;
    int upper_count = 0; // æ˜¯å¦å‘ä¸Šè¿›ä½

    while (true) {
      int sum_val = upper_count;
      bool is_finish = false;
      if ((l1 == nullptr) && (l2 == nullptr)) {
        is_finish = true;
      }
      if (l1 != nullptr) {
        sum_val += l1->val;
        l1 = l1->next;
      }
      if (l2 != nullptr) {
        sum_val += l2->val;
        l2 = l2->next;
      }

      if (is_finish && sum_val == 0)
        break;
      curr->next = new ListNode(sum_val % 10);
      curr = curr->next;
      upper_count = sum_val >= 10 ? 1 : 0;
    }

    return dummy->next;
  }
};

                              #include <array>
#include <map>
#include <vector>
using namespace std;

class Solution {
public:
  bool isNumber(string s) {
    vector<map<char, int>> vmRes{
        {{' ', 0}, {'s', 1}, {'d', 2}, {'.', 4}},
        {{'d', 2}, {'.', 4}},                     // 1. 'sign' before e
        {{'d', 2}, {'.', 3}, {'e', 5}, {' ', 8}}, // 2 digit before dot
        {{'d', 3}, {'e', 5}, {' ', 8}},           // 3 digit after dot
        {{'d', 3}},           // 4 digit after dot, 'blank' before dot
        {{'s', 6}, {'d', 7}}, // 5 e
        {{'d', 7}},           // 6 sign after e
        {{'d', 7}, {' ', 8}}, // 7 digit after e
        {{' ', 8}}            // 8. end with blank
    };

    int p = 0;
    char t = ' ';
    for (char c : s) {
      if ('0' <= c && c <= '9') {
        t = 'd';
      } else if (c == '+' || c == '-') {
        t = 's';
      } else if (c == 'e' || c == 'E') {
        t = 'e';
      } else if (c == '.' || c == ' ') {
        t = c;
      } else {
        t = '?';
      }

      if (vmRes[p].find(t) == vmRes[p].end()) {
        return false;
      }

      p = vmRes[p][t];
    }

    if (p == 2 || p == 3 || p == 7 || p == 8) {
      return true;
    }

    return false;
  }
};
