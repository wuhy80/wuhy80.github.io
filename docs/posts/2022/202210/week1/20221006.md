# 2022-10-06 LeetCode

## 剑指 Offer 35. 复杂链表的复制

```cpp
class Solution {
public:
  Node *copyRandomList(Node *head) {
    map<Node *, Node *> orig_map;
    Node *p = head;
    while (p != nullptr) {
      if (orig_map.find(p) == orig_map.end()) {
        orig_map[p] = new Node(p->val);
      }

      p = p->next;
    }

    p = head;
    while (p != nullptr) {
      if (p->next != nullptr) {
        orig_map[p]->next = orig_map[p->next];
      }

      if (p->random != nullptr) {
        orig_map[p]->random = orig_map[p->random];
      }

      p = p->next;
    }

    return orig_map[head];
  }
};
```
